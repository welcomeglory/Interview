  스택 - Last In Last Out

스택개념이라는 개념을 코드로 짜보세요.
import java.util.EmptyStackException;

    public class Stack {
    // 스택의 요소를 저장할 배열
    private int[] stackArray;
    // 스택의 현재 크기를 나타내는 변수
    private int top;
    // 스택의 용량을 나타내는 변수
    private int capacity;

    // 스택 생성자
    public Stack(int size) {
        // 스택 용량 설정
        capacity = size;
        // 스택 요소를 저장할 배열 생성
        stackArray = new int[capacity];
        // 스택이 비어 있으므로 top을 -1로 초기화
        top = -1;
    }

    // 스택에 요소를 추가하는 메서드
    public void push(int item) {
        // 스택이 가득 찼는지 확인
        if (isFull()) {
            System.out.println("스택이 가득 찼습니다. 요소를 추가할 수 없습니다.");
            return;
        }
        // 요소를 배열에 추가하고 top을 증가
        stackArray[++top] = item;
    }

    // 스택에서 요소를 제거하는 메서드
    public int pop() {
        // 스택이 비어 있는지 확인
        if (isEmpty()) {
            throw new EmptyStackException();
        }
        // top의 요소를 반환하고 top을 감소
        return stackArray[top--];
    }

    // 스택의 가장 위에 있는 요소를 확인하는 메서드
    public int peek() {
        // 스택이 비어 있는지 확인
        if (isEmpty()) {
            throw new EmptyStackException();
        }
        // top의 요소를 반환
        return stackArray[top];
    }

    // 스택이 비어 있는지 확인하는 메서드
    public boolean isEmpty() {
        return top == -1;
    }

    // 스택이 가득 찼는지 확인하는 메서드
    public boolean isFull() {
        return top == capacity - 1;
    }

    // 스택의 크기를 반환하는 메서드
    public int size() {
        return top + 1;
    }

    // 스택의 내용을 출력하는 메서드
    public void printStack() {
        if (isEmpty()) {
            System.out.println("스택이 비어 있습니다.");
        } else {
            System.out.print("스택 요소: ");
            for (int i = 0; i <= top; i++) {
                System.out.print(stackArray[i] + " ");
            }
            System.out.println();
        }
    }

    // 테스트 메인 메서드
    public static void main(String[] args) {
        // 크기가 5인 스택 생성
        Stack stack = new Stack(5);

        // 스택에 요소 추가
        stack.push(1);
        stack.push(2);
        stack.push(3);
        stack.push(4);
        stack.push(5);

        // 스택 출력
        stack.printStack();

        // 스택에서 요소 제거 및 출력
        System.out.println("pop: " + stack.pop());
        System.out.println("pop: " + stack.pop());

        // 스택의 현재 상태 출력
        stack.printStack();

        // 스택의 가장 위에 있는 요소 확인
        System.out.println("peek: " + stack.peek());

        // 스택이 비었는지 확인
        System.out.println("스택이 비었나요? " + stack.isEmpty());

        // 스택에 요소 추가
        stack.push(6);
        stack.printStack();

        // 스택의 크기 확인
        System.out.println("스택의 크기: " + stack.size());
    }
}

  
  큐 - FIFO
  자바스크립트 이벤트루프 큐로 구성

    public class Queue {
    // 큐의 요소를 저장할 배열
    private int[] queueArray;
    // 큐의 앞(front)와 뒤(rear)를 나타내는 변수
    private int front;
    private int rear;
    // 큐의 현재 크기를 나타내는 변수
    private int size;
    // 큐의 용량을 나타내는 변수
    private int capacity;

    // 큐 생성자
    public Queue(int size) {
        // 큐 용량 설정
        capacity = size;
        // 큐 요소를 저장할 배열 생성
        queueArray = new int[capacity];
        // 큐가 비어 있으므로 front와 rear를 -1로 초기화
        front = -1;
        rear = -1;
        // 큐 크기를 0으로 초기화
        this.size = 0;
    }

    // 큐에 요소를 추가하는 메서드
    public void enqueue(int item) {
        // 큐가 가득 찼는지 확인
        if (isFull()) {
            System.out.println("큐가 가득 찼습니다. 요소를 추가할 수 없습니다.");
            return;
        }
        // 큐가 비어 있을 경우 front를 0으로 설정
        if (isEmpty()) {
            front = 0;
        }
        // rear를 증가시키고 요소를 배열에 추가
        rear = (rear + 1) % capacity;
        queueArray[rear] = item;
        // 큐 크기를 증가
        size++;
    }

    // 큐에서 요소를 제거하는 메서드
    public int dequeue() {
        // 큐가 비어 있는지 확인
        if (isEmpty()) {
            System.out.println("큐가 비어 있습니다. 요소를 제거할 수 없습니다.");
            return Integer.MIN_VALUE;
        }
        // front의 요소를 반환
        int item = queueArray[front];
        // front를 증가시키고 큐 크기를 감소
        front = (front + 1) % capacity;
        size--;
        // 큐가 비어 있을 경우 front와 rear를 -1로 초기화
        if (isEmpty()) {
            front = -1;
            rear = -1;
        }
        return item;
    }

    // 큐의 가장 앞에 있는 요소를 확인하는 메서드
    public int peek() {
        // 큐가 비어 있는지 확인
        if (isEmpty()) {
            System.out.println("큐가 비어 있습니다.");
            return Integer.MIN_VALUE;
        }
        // front의 요소를 반환
        return queueArray[front];
    }

    // 큐가 비어 있는지 확인하는 메서드
    public boolean isEmpty() {
        return size == 0;
    }

    // 큐가 가득 찼는지 확인하는 메서드
    public boolean isFull() {
        return size == capacity;
    }

    // 큐의 크기를 반환하는 메서드
    public int size() {
        return size;
    }

    // 큐의 내용을 출력하는 메서드
    public void printQueue() {
        if (isEmpty()) {
            System.out.println("큐가 비어 있습니다.");
        } else {
            System.out.print("큐 요소: ");
            for (int i = 0; i < size; i++) {
                System.out.print(queueArray[(front + i) % capacity] + " ");
            }
            System.out.println();
        }
    }

    // 테스트 메인 메서드
    public static void main(String[] args) {
        // 크기가 5인 큐 생성
        Queue queue = new Queue(5);

        // 큐에 요소 추가
        queue.enqueue(1);
        queue.enqueue(2);
        queue.enqueue(3);
        queue.enqueue(4);
        queue.enqueue(5);

        // 큐 출력
        queue.printQueue();

        // 큐에서 요소 제거 및 출력
        System.out.println("dequeue: " + queue.dequeue());
        System.out.println("dequeue: " + queue.dequeue());

        // 큐의 현재 상태 출력
        queue.printQueue();

        // 큐의 가장 앞에 있는 요소 확인
        System.out.println("peek: " + queue.peek());

        // 큐가 비었는지 확인
        System.out.println("큐가 비었나요? " + queue.isEmpty());

        // 큐에 요소 추가
        queue.enqueue(6);
        queue.printQueue();

        // 큐의 크기 확인
        System.out.println("큐의 크기: " + queue.size());
    }
  }
